		}

			}
		}

			multiply_temp_variable = { manpower_in_state = 20 }
		}
	}
	if = {
		limit = {
			ROOT = {
				has_idea = scraping_the_barrel
			}
		}
		THIS = {

			multiply_temp_variable = { manpower_in_state = 25 }
		}
	}
	THIS = {
		set_temp_variable = { civilians_in_state = THIS:state_population_k }
		subtract_from_variable = { civilians_in_state = manpower_in_state }
		set_temp_variable = { children_in_state = civilians_in_state }
		divide_temp_variable = { children_in_state = 100 }
		multiply_temp_variable = { children_in_state = 6 }

		dhondt_compare_parties = yes

		dhondt_first_seat = yes
	}
