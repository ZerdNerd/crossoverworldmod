middleage_to_preindustrial = {
	set_technology = {
		gw_artillery = 1
		early_fighter = 1
	}
	if = {
		limit = {
			not = { has_dlc = "Man the Guns" }
		}
		set_technology = {
			early_destroyer = 1
			early_light_cruiser = 1
			transport = 1
		}
	}
	if = {
		limit = {
			has_dlc = "Man the Guns"
		}
		set_technology = {
			early_ship_hull_light = 1
			early_ship_hull_cruiser = 1
			mtg_transport = 1
			basic_battery = 1
			basic_light_battery = 1
			basic_cruiser_armor_scheme = 1
		}	
	}
}
