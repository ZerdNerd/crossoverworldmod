	option = { #Avalor is truly valor.
		name = truce_proposal.2.a
		hidden_effect = {
			every_country = {
				limit = {
					is_in_faction_with = AVA
				}
				set_country_flag = avalor_signed_peace_with_honor
			}
		}
		AVA = {
			leave_faction = yes
		}
		white_peace = AVA
		every_country = {
			limit = {
				is_in_faction_with = ROOT
			}

		}

			limit = {
				NOT = {
					tag = EN3

		}

			set_country_flag = signed_peace_with_honor

				}
